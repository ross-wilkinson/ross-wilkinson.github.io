# Power


Create subplot for power v mass



```matlab:Code
ax1 = subplot(3,1,1);
x2 = 0:0.1:35;

for i = 1:19
    switch i
        case{12,16}
        otherwise
            wgt = mdl{i}.data.wgt;
            pwr = mdl{i}.data.pwr;
            x1 = min(wgt):0.1:max(wgt);
            [fitPwr, gofPwr] = fit(wgt,pwr,"poly2");
            plot(x1,fitPwr(x1),'k')
            xlim([0 35])
            ylim([0 2000])
            hold on
            scatter(wgt,pwr,10,'k','filled')
            r2(i) = gofPwr.rsquare;
    end
end

xlabel('Hanging mass (kg)')
ylabel('Maximal 1-s power output (W)')
xlim([0 35])
ylim([0 2000])
ax1.Box = 'off';
ax1.XTick = x2(1:50:end);
ax1.XTickLabels = round(x2(1:50:end)/2.205,1);
title('A')
ax1.Title.HorizontalAlignment = 'left';
ax1.Title.VerticalAlignment = 'middle';
ax1.Title.Position = [2, ax1.YLim(2), 0];

```



Plot Pmax and Mopt line



```matlab:Code
for i = 1:19
    switch i
        case{12,16}
            P_max(i) = nan;
            M_hopt(i) = nan;
        otherwise

            line([I_max(i)/10 I_max(i)/10],[0 P_max(i)],'color',c2,'linestyle','--')
    end
end

```



Plot power fit stats



```matlab:Code
mean_r2 = string(round(mean(rmoutliers(r2),'omitnan'),2));
```

# Torque


Create subplot for torque v mass



```matlab:Code
ax3 = subplot(3,1,3);
x2 = 0:0.1:20;

for i = 1:19
    switch i
        case{12,16}
            r2(i) = nan;
            p1(i) = nan;
        otherwise
            wgt = mdl{i}.data.wgt;
            trq = mdl{i}.data.trq;
            x1 = min(wgt):0.1:max(wgt);
            [fitTrq, gofTrq] = fit(wgt,trq,"poly1","Lower",[-Inf 0],"Upper",[Inf 0]);
%             plot(x1,fitTrq(x1),'k')
            plot(x1,x1*4.5,'k')
            xlim([0 35])
            ylim([0 300])
            hold on
            scatter(wgt,trq,10,'k','filled')
            r2(i) = gofTrq.rsquare;
            p1(i) = fitTrq.p1;
    end
end

xlabel('Hanging mass (kg)')
ylabel('Crank torque (N\cdotm)')
xlim([0 35])
ylim([0 300])

ax3.XTick = x2(1:50:end);
```
